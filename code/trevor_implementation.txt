"""
Fills bin based on remaining empty space (x).
"""
def optimizeCapacity(item, x):
    if item.weight == x
        insert item into bin
        remove item from a
    elif item.weight < x
        optimizeCapacity(item, x-1)

"""
Creates all needed bins for items that are over half the value of the max weight (w)
"""
def createBins(a, w)
    for item in a:
        if item.weight > w:
